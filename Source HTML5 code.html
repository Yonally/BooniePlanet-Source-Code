var arrowDictionary = {}; 


var timeoutPerArrow = {}; 


function createArrow(id, style, rotated, delay) 


{ 


if(id in arrowDictionary) return; 






var view = !rotated ? "img/arrow.png" : "img/arrow_02.png"; 


var div = document.createElement('div'); 


div.innerHTML = '<img src="' + view + '" class="bounce" alt="" style="position: absolute;' + style + ';z-index: 100000;">'; 


div.setAttribute('class', 'arrow_pointer'); 


div.setAttribute('id', id); 


document.body.appendChild(div); 




if(delay != 0) 


{ 


$("#"+id).hide(); 




var timeout = setTimeout( 


function() 


{ 


$("#"+id).fadeIn(1000); 


}, 


delay); 




timeoutPerArrow[id] = timeout; 


} 






arrowDictionary[id] = div; 


} 




function hideArrow(id) 


{ 


if(id in arrowDictionary) 


{ 


$("#"+id).fadeOut( 


500, 


function() 


{ 


$(this).remove() 


} 


); 




delete arrowDictionary[id]; 


} 


} 




function hideAllArrows() 


{ 


for(id in timeoutPerArrow) 


{ 


clearTimeout(timeoutPerArrow[id]); 


delete timeoutPerArrow[id]; 


} 




for(id in arrowDictionary) 


{ 


$("#"+id).remove(); 


delete arrowDictionary[id]; 


} 


}

function getFlashVersion() 


{ 


var pluginVersion = getFlashVersionFromNavigator(); 




if(pluginVersion == undefined) 


{ 


pluginVersion = getFlashVersionFromMimeTypes(); 


} 




return pluginVersion; 


} 




function getFlashVersionFromNavigator() 


{ 


var flash = navigator.plugins.namedItem('Shockwave Flash'); 


var typeOf = typeof flash; 




if (typeOf != 'object' || flash == null || flash == undefined) 


{ 


// flash is not present 


return undefined; 


} 




if(flash.version) 


{ 


return flash.version; 


} 


else 


{ 


//No version property (e.g. in Chrome) 


return flash.description.replace(/Shockwave Flash /,""); 


} 


} 




function getFlashVersionFromMimeTypes() 


{ 


if (navigator.mimeTypes && navigator.mimeTypes.length) 


{ 


var mimeType = navigator.mimeTypes['application/x-shockwave-flash']; 


hasFlash = mimeType && mimeType.enabledPlugin; 


if (hasFlash) 


{ 


return mimeType.enabledPlugin.description; 


} 


} 




return undefined; 


}
